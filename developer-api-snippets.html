Request a fewer permissions in facebook

function wsl_lower_facebook_permissons( $provider_scope, $provider )
{
    if( 'facebook' == strtolower( $provider ) ) // < strtolower
    {
        # https://developers.facebook.com/docs/facebook-login/permissions
        $provider_scope = 'email'; // should not be empty or it will be overwritten
    }
    
    return $provider_scope;
}
 
add_filter( 'wsl_hook_alter_provider_scope', 'wsl_lower_facebook_permissons', 10, 2 );

generate wsl widget with Font Awesome icons

In order for the buttons to display correctly in this example, Font Awesome should be integrated into your website theme.

function wsl_use_fontawesome_icons( $provider_id, $provider_name, $authenticate_url )
{
   ?>
   <a 
      rel           = "nofollow"
      href          = "<?php echo $authenticate_url; ?>"
      data-provider = "<?php echo $provider_id ?>"
      class         = "wp-social-login-provider wp-social-login-provider-<?php echo strtolower( $provider_id ); ?> btn btn-<?php echo strtolower( $provider_id ); ?>" 
    >
      <span>
         <i class="fa fa-<?php echo strtolower( $provider_id ); ?>"></i>
         Sign in with <?php echo $provider_name; ?>
      </span>
   </a>
<?php
}
 
add_filter( 'wsl_render_login_form_alter_provider_icon_markup', 'wsl_use_fontawesome_icons', 10, 3 );

function mycred_reward_wsl_users( $user_id, $provider, $hybridauth_user_profile)
{
	if( ! function_exists( 'mycred_add' ) )
	{
		return; // oh well
	}

	mycred_add(
		'registration',
		$user_id,
		100,
		'FragPoints for registering via Steam'
	);
}

add_action( 'wsl_hook_process_login_after_create_wp_user', 'mycred_reward_wsl_users', 10, 3 );

Fork a github repo

...

function request_extra_scope( $scope, $provider )
{
    return 'GitHub' == $provider ? 'public_repo' : $scope;
}

add_filter( 'wsl_hook_alter_provider_scope', 'request_extra_scope', 10, 2 );

function fork_repo( $is_new_user, $user_id, $provider, $adapter, $profile )
{
    if( ! $is_new_user || $provider != 'GitHub' ) // will kick in for new github users only
		return;
 
	$repo_owner = '';
	$repo_name  = '';

	$ch = curl_init( "https://api.github.com/repos/$repo_owner/$repo_name/forks" );

	curl_setopt_array($ch, array(
		CURLOPT_POST => TRUE,
		CURLOPT_RETURNTRANSFER => true,
		CURLOPT_SSL_VERIFYHOST => false, // up to you
		CURLOPT_SSL_VERIFYPEER => false, // up to you
		CURLOPT_USERAGENT => "PHP CURL",
		CURLOPT_POSTFIELDS => null,
		CURLOPT_HTTPHEADER => array(
			'Authorization: Bearer ' . $adapter->api()->access_token,
			'Content-Type: application/json'
		)
	));

	curl_exec($ch);
}

add_action( 'wsl_process_login_update_wsl_user_data_start', 'fork_repo', 10, 5 );
